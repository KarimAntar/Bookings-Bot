# Google Cloud Bookings Bot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build and deploy a production-ready TypeScript/Bun Slack Socket Mode bot that reviews booking screenshots through Gemini and runs continuously under systemd on the existing Google Compute Engine `e2-micro` VM.

**Architecture:** Slack Bolt receives message events through an outbound Socket Mode connection. Focused adapters validate and download Slack files, a provider-independent review service sends images to Gemini, Zod validates every model response, and a conservative fallback converts uncertainty into `needs_human_review`. A private GitHub repository supplies the VM, while systemd supervises the Bun process and loads secrets from a root-managed environment file.

**Tech Stack:** Bun, strict TypeScript, `@slack/bolt`, `@google/genai`, Zod, Pino, Biome, Bun test, Bash, systemd, Google Compute Engine.

Create `src/runtime/dedupe-cache.ts`:

```ts
export class DedupeCache {
  private readonly entries = new Map<string, number>();

  constructor(
    private readonly capacity: number,
    private readonly ttlMs: number,
    private readonly now: () => number = Date.now,
  ) {}

  claim(key: string): boolean {
    const timestamp = this.now();
    this.prune(timestamp);
    if (this.entries.has(key)) return false;

    while (this.entries.size >= this.capacity) {
      const oldest = this.entries.keys().next().value as string | undefined;
      if (oldest === undefined) break;
      this.entries.delete(oldest);
    }

    this.entries.set(key, timestamp + this.ttlMs);
    return true;
  }

  private prune(timestamp: number): void {
    for (const [key, expiresAt] of this.entries) {
      if (expiresAt <= timestamp) this.entries.delete(key);
    }
  }
}
```

Create `src/runtime/semaphore.ts`:

```ts
export class QueueFullError extends Error {
  constructor() {
    super("Review queue is full");
    this.name = "QueueFullError";
  }
}

type Release = () => void;

export class Semaphore {
  private active = 0;
  private readonly queue: Array<(release: Release) => void> = [];

  constructor(
    private readonly limit: number,
    private readonly maxQueue: number,
  ) {}

  acquire(): Promise<Release> {
    if (this.active < this.limit) {
      this.active += 1;
      return Promise.resolve(this.createRelease());
    }

    if (this.queue.length >= this.maxQueue) {
      return Promise.reject(new QueueFullError());
    }

    return new Promise((resolve) => this.queue.push(resolve));
  }

  async run<T>(work: () => Promise<T>): Promise<T> {
    const release = await this.acquire();
    try {
      return await work();
    } finally {
      release();
    }
  }

  private createRelease(): Release {
    let released = false;
    return () => {
      if (released) return;
      released = true;
      const next = this.queue.shift();
      if (next) next(this.createRelease());
      else this.active -= 1;
    };
  }
}
```

- [ ] **Step 5: Implement timeout and retry helpers**

Create `src/runtime/retry.ts`:

```ts
export class TimeoutError extends Error {
  constructor(timeoutMs: number) {
    super(`Operation timed out after ${timeoutMs}ms`);
    this.name = "TimeoutError";
  }
}

export async function withTimeout<T>(promise: Promise<T>, timeoutMs: number): Promise<T> {
  let timer: ReturnType<typeof setTimeout> | undefined;
  const timeout = new Promise<never>((_, reject) => {
    timer = setTimeout(() => reject(new TimeoutError(timeoutMs)), timeoutMs);
  });

  try {
    return await Promise.race([promise, timeout]);
  } finally {
    if (timer) clearTimeout(timer);
  }
}

interface RetryOptions {
  readonly attempts: number;
  readonly baseDelayMs: number;
  readonly isTransient: (error: unknown) => boolean;
  readonly sleep?: (delayMs: number) => Promise<void>;
  readonly random?: () => number;
}

export async function withRetry<T>(work: () => Promise<T>, options: RetryOptions): Promise<T> {
  const sleep = options.sleep ?? ((delayMs) => Bun.sleep(delayMs));
  const random = options.random ?? Math.random;
  let lastError: unknown;

  for (let attempt = 1; attempt <= options.attempts; attempt += 1) {
    try {
      return await work();
    } catch (error) {
      lastError = error;
      if (attempt === options.attempts || !options.isTransient(error)) throw error;
      const backoff = options.baseDelayMs * 2 ** (attempt - 1);
      await sleep(backoff + Math.floor(backoff * 0.25 * random()));
    }
  }

  throw lastError;
}
```

```ts
import type { ReviewRequest } from "../domain/review-request";
import { ReviewResultSchema, type ReviewResult } from "../domain/review-result";
import { withRetry, withTimeout } from "../runtime/retry";
import { BOOKING_REVIEW_POLICY, REVIEW_RESPONSE_JSON_SCHEMA } from "../reviews/booking-policy";

export type GeminiGenerate = (request: ReviewRequest) => Promise<string>;

export class GeminiProvider implements AIProvider {
  constructor(
    private readonly generate: GeminiGenerate,
    private readonly timeoutMs: number,
  ) {}

  async review(request: ReviewRequest): Promise<ReviewResult> {
    const text = await withRetry(
      () => withTimeout(this.generate(request), this.timeoutMs),
      {
        attempts: 3,
        baseDelayMs: 500,
        isTransient: isTransientGeminiError,
      },
    );

    let raw: unknown;
    try {
      raw = JSON.parse(text);
    } catch {
      throw new Error("Gemini returned invalid JSON");
    }

    return ReviewResultSchema.parse(raw);
  }
}

export function createGeminiGenerate(
  apiKey: string,
  model: string,
): GeminiGenerate {
  const client = new GoogleGenAI({ apiKey });

  return async (request) => {
    const parts = [
      {
        text: `${BOOKING_REVIEW_POLICY}\nMessage context: ${request.messageText || "None provided."}`,
      },
      ...request.images.map((image) => ({
        inlineData: {
          data: Buffer.from(image.data).toString("base64"),
          mimeType: image.mimeType,
        },
      })),
    ];

    const response = await client.models.generateContent({
      model,
      contents: [{ role: "user", parts }],
      config: {
        responseMimeType: "application/json",
        responseJsonSchema: REVIEW_RESPONSE_JSON_SCHEMA,
      },
    });

    if (!response.text) throw new Error("Gemini returned an empty response");
    return response.text;
  };
}

export function isTransientGeminiError(error: unknown): boolean {
  if (!(error instanceof Error)) return false;
  const status = (error as Error & { status?: number }).status;
  return status === 408 || status === 429 || (status !== undefined && status >= 500);
}
```

```ts
export interface SlackFileLike {
  readonly id: string;
  readonly name?: string;
  readonly mimetype?: string;
  readonly size?: number;
  readonly url_private_download?: string;
}

export interface AcceptedSlackImage {
  readonly id: string;
  readonly name: string;
  readonly mimeType: "image/png" | "image/jpeg" | "image/webp";
  readonly size: number;
  readonly downloadUrl: string;
}

const supportedTypes = new Set(["image/png", "image/jpeg", "image/webp"]);

export function validateSlackFiles(
  files: readonly SlackFileLike[],
  limits: { readonly maxAttachments: number; readonly maxImageBytes: number },
): { accepted: AcceptedSlackImage[]; flags: string[] } {
  const accepted: AcceptedSlackImage[] = [];
  const flags: string[] = [];

  if (files.length > limits.maxAttachments) flags.push("too_many_attachments");

  for (const file of files.slice(0, limits.maxAttachments)) {
    if (!file.mimetype?.startsWith("image/")) continue;
    if (!supportedTypes.has(file.mimetype)) {
      flags.push("unsupported_image_type");
      continue;
    }
    if (!file.size || file.size > limits.maxImageBytes) {
      flags.push("image_too_large");
      continue;
    }
    if (!file.url_private_download) {
      flags.push("missing_download_url");
      continue;
    }

    accepted.push({
      id: file.id,
      name: file.name ?? file.id,
      mimeType: file.mimetype as AcceptedSlackImage["mimeType"],
      size: file.size,
      downloadUrl: file.url_private_download,
    });
  }

  return { accepted, flags: [...new Set(flags)] };
}
```

```ts
import type { ReviewImage } from "../domain/review-request";
import type { AcceptedSlackImage } from "./file-policy";

export async function downloadSlackImage(
  file: AcceptedSlackImage,
  botToken: string,
  timeoutMs: number,
  maxBytes: number,
  fetcher: typeof fetch = fetch,
): Promise<ReviewImage> {
  const response = await fetcher(file.downloadUrl, {
    headers: { authorization: `Bearer ${botToken}` },
    signal: AbortSignal.timeout(timeoutMs),
    redirect: "follow",
  });

  if (!response.ok) throw new Error(`Slack file download failed with HTTP ${response.status}`);
  if (!response.body) throw new Error("Slack file download returned no body");

  const reader = response.body.getReader();
  const chunks: Uint8Array[] = [];
  let total = 0;

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    total += value.byteLength;
    if (total > maxBytes) {
      await reader.cancel();
      throw new Error("Slack image exceeded size limit");
    }
    chunks.push(value);
  }

  const data = new Uint8Array(total);
  let offset = 0;
  for (const chunk of chunks) {
    data.set(chunk, offset);
    offset += chunk.byteLength;
  }

  return { id: file.id, name: file.name, mimeType: file.mimeType, data };
}
```

```ts
import type { AIProvider } from "../ai/ai-provider";
import type { ReviewRequest } from "../domain/review-request";
import { humanReviewFallback, type ReviewResult } from "../domain/review-result";

export class ReviewService {
  constructor(
    private readonly provider: AIProvider,
    private readonly lowConfidenceThreshold: number,
  ) {}

  async review(request: ReviewRequest): Promise<ReviewResult> {
    if (request.images.length === 0) return humanReviewFallback("no_readable_images");

    try {
      const result = await this.provider.review(request);
      if (result.decision === "approved" && result.confidence < this.lowConfidenceThreshold) {
        return {
          ...result,
          decision: "needs_human_review",
          reasoning: "The automated review confidence was below the approval threshold.",
          flags: [...new Set([...result.flags, "low_confidence"])],
        };
      }
      return result;
    } catch {
      return humanReviewFallback("provider_error");
    }
  }
}
```

```ts
import type { ReviewImage, ReviewRequest } from "../domain/review-request";
import type { ReviewResult } from "../domain/review-result";
import { DedupeCache } from "../runtime/dedupe-cache";
import { QueueFullError, Semaphore } from "../runtime/semaphore";
import { formatReview, terminalReaction } from "./format-review";
import { type AcceptedSlackImage, type SlackFileLike, validateSlackFiles } from "./file-policy";

interface SlackMessage {
  readonly channel: string;
  readonly ts: string;
  readonly text?: string;
  readonly subtype?: string;
  readonly bot_id?: string;
  readonly files?: readonly SlackFileLike[];
}

interface SlackClientLike {
  readonly reactions: {
    add(input: { channel: string; timestamp: string; name: string }): Promise<unknown>;
    remove(input: { channel: string; timestamp: string; name: string }): Promise<unknown>;
  };
  readonly chat: {
    postMessage(input: { channel: string; thread_ts: string; text: string }): Promise<unknown>;
  };
}

interface ListenerDependencies {
  readonly config: {
    readonly allowedChannelIds: ReadonlySet<string>;
    readonly maxAttachments: number;
    readonly maxImageBytes: number;
    readonly slackBotToken: string;
    readonly downloadTimeoutMs: number;
  };
  readonly dedupe: DedupeCache;
  readonly gate: Semaphore;
  readonly review: (request: ReviewRequest) => Promise<ReviewResult>;
  readonly download: (file: AcceptedSlackImage) => Promise<ReviewImage>;
  readonly log?: (entry: Record<string, unknown>, message: string) => void;
}

export class MessageListener {
  constructor(private readonly dependencies: ListenerDependencies) {}

  async handle(input: {
    readonly eventId: string;
    readonly message: SlackMessage;
    readonly client: SlackClientLike;
  }): Promise<void> {
    const { eventId, message, client } = input;
    if (message.bot_id) return;
    if (message.subtype && message.subtype !== "file_share") return;
    if (
      this.dependencies.config.allowedChannelIds.size > 0 &&
      !this.dependencies.config.allowedChannelIds.has(message.channel)
    ) return;
    if (!message.files?.length || !this.dependencies.dedupe.claim(eventId)) return;

    await this.safeReaction(client, "add", message, "eyes");

    try {
      await this.dependencies.gate.run(async () => {
        const validation = validateSlackFiles(message.files ?? [], this.dependencies.config);
        const images = await Promise.all(
          validation.accepted.map((file) => this.dependencies.download(file)),
        );
        const result = await this.dependencies.review({
          eventId,
          messageText: message.text ?? "",
          images,
        });
        const finalResult = validation.flags.length
          ? {
              ...result,
              decision: "needs_human_review" as const,
              reasoning: "One or more attachments could not be reviewed safely.",
              flags: [...new Set([...result.flags, ...validation.flags])],
            }
          : result;

        await client.chat.postMessage({
          channel: message.channel,
          thread_ts: message.ts,
          text: formatReview(finalResult),
        });
        await this.safeReaction(client, "remove", message, "eyes");
        await this.safeReaction(client, "add", message, terminalReaction(finalResult.decision));
        this.dependencies.log?.({ eventId, decision: finalResult.decision }, "review_completed");
      });
    } catch (error) {
      const queueFull = error instanceof QueueFullError;
      await client.chat.postMessage({
        channel: message.channel,
        thread_ts: message.ts,
        text: formatReview({
          decision: "needs_human_review",
          reasoning: queueFull
            ? "The automated review queue is full. Please review this booking manually."
            : "Automatic review could not be completed safely.",
          confidence: 0,
          extractedFields: {},
          flags: [queueFull ? "queue_full" : "processing_error"],
        }),
      });
      this.dependencies.log?.({ eventId, errorType: queueFull ? "queue_full" : "processing_error" }, "review_failed");
    }
  }

  private async safeReaction(
    client: SlackClientLike,
    action: "add" | "remove",
    message: SlackMessage,
    name: string,
  ): Promise<void> {
    try {
      await client.reactions[action]({ channel: message.channel, timestamp: message.ts, name });
    } catch {
      // Reactions are optional and must not block the review reply.
    }
  }
}
```

```ts
import { parseEnv } from "./config/env";

async function main(): Promise<void> {
  const config = parseEnv(process.env);
  const { app, logger } = createSlackApp(config);

    if (stopping) return;
    stopping = true;

    const forceExit = setTimeout(() => process.exit(1), 25_000);
    forceExit.unref();
    await app.stop();
    clearTimeout(forceExit);
    logger.info({ event: "bot_stopped" }, "bot_stopped");
    process.exit(0);
  };

  process.once("SIGTERM", () => void stop("SIGTERM"));
  process.once("SIGINT", () => void stop("SIGINT"));

  await app.start();
  logger.info(
    { event: "bot_started", model: config.geminiModel, socketMode: true },
    "bot_started",
  );
}

  const message = error instanceof Error ? error.message : "Unknown startup error";
  console.error(JSON.stringify({ level: "fatal", event: "startup_failed", message }));
  process.exit(1);
});
```
